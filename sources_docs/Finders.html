var utils = require('./Utils')
var executeSingle = utils.executeSingle
var executeMany = utils.executeMany
var Promise = require('bluebird')
var Chance = require('chance')
var chance = new Chance()
var _ = require('lodash')
var QueryBuilder = require('./QueryBuilder')

var findById = function(model, id, selectList, embedList) {
	var queryBuilder = new QueryBuilder()
	if(!isNaN(id)) id = parseInt(id)
	queryBuilder.model(model).id(id)
	if(selectList) queryBuilder.selectList(selectList)
	if(embedList) queryBuilder.embedList(embedList)
	var query = queryBuilder.getQuery()
	return executeSingle(model, query)
}

var findByIdArray = function(model, idArray, selectList, embedList) {
	var queryBuilder = new QueryBuilder()
	queryBuilder.model(model).idArray(idArray)
	if(selectList) queryBuilder.selectList(selectList)
	if(embedList) queryBuilder.embedList(embedList)
	var query = queryBuilder.getQuery()
	return executeMany(model, query)
}

var findAll = function(model, selectList, embedList) {
	var queryBuilder = new QueryBuilder()
	queryBuilder.model(model)
	if(selectList) queryBuilder.selectList(selectList)
	if(embedList) queryBuilder.embedList(embedList)
	var query = queryBuilder.getQuery()
	return executeMany(model, query)
}

var findByExample = function(model, example, selectList, embedList) {
	var queryBuilder = new QueryBuilder()
	queryBuilder.model(model)
	if(selectList) queryBuilder.selectList(selectList)
	if(embedList) queryBuilder.embedList(embedList)
	Object.keys(example).forEach(function(key){
		var clause = {
			operator: 'eq',
			field: key,
			value: example[key]
		}
		queryBuilder.addWhere(clause)
	})
	var query = queryBuilder.getQuery()
	return executeMany(model, query)
}

var findByQueryBuilder = function(model, queryBuilder) {
	var query = queryBuilder.model(model).getQuery()
	return executeMany(model, query)
}

var findByQueryString = function(model, query) {
	return executeMany(model, query)
}

var count = function(model) {
	var queryBuilder = new QueryBuilder().model(model).isCount(true)
	return model.db().queryOneScalarAsync(queryBuilder.getQuery(), {})
}

var findRandom = function(model, number, selectList, embedList) {
	var deferred = Promise.defer()
	if(_.isUndefined(number) || number == 0) deferred.reject('Please supply findRandom with a number of records you want to get back. >0')
	var countPromise = count(model)
	countPromise
		.then(function(count){
			if(count<1) deferred.reject('There are no models of type: ' + model.label + ' in the database.')
			var random = chance.integer({min: 0, max: count-number})
			var queryBuilder = new QueryBuilder().limit(number).skip(random)
			if(selectList) queryBuilder.selectList(selectList)
			if(embedList) queryBuilder.embedList(embedList)
			model.findByQueryBuilder(queryBuilder)
				.then(function(models){
					deferred.resolve(models)
				})
				.catch(function(err){
					deferred.reject(err)
				})
		})
		.catch(function(err){
			deferred.reject(err)
		})
	return deferred.promise
}

module.exports = {
	findById: findById,
	findByIdArray: findByIdArray,
	findAll: findAll,
	findByExample: findByExample,
	findByQueryBuilder: findByQueryBuilder,
	findByQueryString: findByQueryString,
	count: count,
	findRandom: findRandom
}
