var Constants = require('./Constants')
var Promise = require('bluebird')
var _ = require('lodash')
var FieldTypes = require('./FieldTypes')

var validateModel = function(model) {
	if(model.getEvents()[Constants.Events.preValidation])
		model.getEvents()[Constants.Events.preValidation].run(model)
	var messages = validateTypes(model)
	if(model.getEvents()[Constants.Events.postValidation])
		model.getEvents()[Constants.Events.postValidation].run(model)
	return messages
}

var validateTypes = function(model) {
	var messages = []
	for (var key in model.data) {
		if (model.data.hasOwnProperty(key)) {
			var result = validateType(model.definition[key].type, model[key])
			if(result) messages.push(result)
		}
	}
	return messages
}
