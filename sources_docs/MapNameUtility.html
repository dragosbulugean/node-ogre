    This module helps keep sanity in map naming in queries.

    var _ = require('lodash')

var Neo4jMapNameUtility = module.exports = function() {
}

Neo4jMapNameUtility.prototype.regulate = function(mapName, number) {
	return this[mapName].split('_')[0] + '_' + (parseInt(this[mapName].split('_')[1])+number)
}

Neo4jMapNameUtility.prototype.getCurrentMapName = function() {
	return this.currentMapName
}

Neo4jMapNameUtility.prototype.getLastMapName = function() {
	if(!this.currentMapName) return
	var lastMapName = this.regulate('currentMapName', -1)
	return lastMapName
}

Neo4jMapNameUtility.prototype.setGetLastMapName = function() {
	if(!this.currentMapName) return
	var lastMapName = this.regulate('currentMapName', -1)
	this.currentMapName = lastMapName;
	return lastMapName
}

Neo4jMapNameUtility.prototype.getNextMapName = function() {
	if(!this.currentMapName)  return 'm_0'
	var nextMapName = this.regulate('currentMapName', +1)
	return nextMapName
}

Neo4jMapNameUtility.prototype.setGetNextMapName = function() {
	if(!this.currentMapName) return this.currentMapName = 'm_0'
	var nextMapName = this.regulate('currentMapName', +1)
	this.currentMapName = nextMapName
	return nextMapName
}

Neo4jMapNameUtility.prototype.getCurrentRelationMapName = function() {
	return this.currentRelationMapName
}

Neo4jMapNameUtility.prototype.getNextRelationMapName = function() {
	if(!this.currentRelationMapName) return 'rm_0'
	var lastRelationMapName = this.regulate('currentRelationMapName', +1)
	return lastRelationMapName;
}

Neo4jMapNameUtility.prototype.setGetNextRelationMapName = function() {
	if(!this.currentRelationMapName) return this.currentRelationMapName = 'rm_0'
	var nextRelationMapName = this.regulate('currentRelationMapName', +1)
	this.currentRelationMapName = nextRelationMapName
	return nextRelationMapName;
}

Neo4jMapNameUtility.prototype.getMapNames = function(current, list) {
	if(!this.currentMapName) return
	if(!list) list = []
	if(!current) current = this.currentMapName
	var nr = parseInt(current.split('_')[1])
	if(nr>=0) {
		list.push(current)
		current = 'm_'+(nr-1)
		this.getMapNames(current,list)
	}
	return list
}

Neo4jMapNameUtility.prototype.getRelationMapNames = function(current, list) {
	if(!this.currentRelationMapName) return
	if(!list) list = []
	if(!current) current = this.currentRelationMapName
	if(!current) return list
	var nr = parseInt(current.split('_')[1])
	if(nr>=0) {
		list.push(current)
		current = 'rm_'+(nr-1)
		this.getRelationMapNames(current,list)
	}
	return list
}

Neo4jMapNameUtility.prototype.getAllUsedMapNames = function() {
	var l1 = this.getMapNames() || []
	var l2 = this.getRelationMapNames() || []
	var value = [].concat(l1).concat(l2) + ','
	return value
}

Neo4jMapNameUtility.prototype.getAllUsedMapNamesWithout = function(mapName) {
	var l1 = this.getMapNames() || []
	l1 = _.difference(l1, [mapName])
	var l2 = this.getRelationMapNames() || []
	l2 = _.difference(l2, [mapName])
	var value = [].concat(l1).concat(l2) + ','
	return value
}

Neo4jMapNameUtility.prototype.reset = function() {
	delete this.currentMapName
	delete this.currentRelationMapName
}
