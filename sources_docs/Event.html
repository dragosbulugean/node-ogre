var Event = function(eventName, eventHandler) {
	this.eventName = eventName
	this.eventHandlers = [eventHandler]
}

Event.prototype.run = function(data) {
	for (var event in this.eventHandlers) {
		this.eventHandlers[event](data)
	}
}

Event.prototype.extendEventBefore = function(fn) {
	this.eventHandlers.unshift(fn)
}

Event.prototype.extendEventAfter = function(fn) {
	this.eventHandlers.push(fn)
}
